%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%- To access the command for the front page, above 
%	\begin{document} environment write:
%	\input{kaufrontpage}% Frontpage for kau thesis
%	to gain access to the command.
%
%
% - Input this command in the top of your \begin{document} (basically copy-paste)
%	 and write your text in the "{}" to provide the command with arguments. 
%	When they are empty like in below example, they'll default to a hard-coded
%	value.
%
%	Copy-paste this:
%	\frontpage%					Sv / Eng
%	{}					% Uppsatsens titel / Thesis title
%	{}					% Uppsatsens undertitel / Thesis subtitle
%	{}					% Uppsatsens engelska titel om ni skriver på svenska / Thesis Swedish title if you're writing in English
%	{}					% Uppsatsens underrubrik på engelska om ni skriver på svenska / Thesis Swedish subtitle if you're writing in English
%	{}					% Första författaren / First author
%	{}					% Andra författaren / Second author. OBS/NOTE: If no second author, leave this field with only a "space" to make it blank.
%	{
%		{}				% Fakultet / Faculty
%		{}				% Ämne / Subject
%		{}				% Nivå - Högskolepoäng / Points
%		{}				% Handledare / Supervisor
%		{}				% Examinator / Examiner
%		{}				% Oppositionsdatum / Opposition date
%		{English}		% Språk för framsidan / Language for frontpage. OBS/NOTE: Writing "English" (case-sensitive) sets frontpage to English. 
%						% Writing anything else sets it to Swedish.
%	}
%
%

% - Using the latex template from 97 for Kau?
%	The \documentclass[]{} takes one option: "twoside" in the template.
% 	Either remove this option or change it to "oneside". When the geometry 
% 	package gets included, the article template will use the correct 
% 	margins from the twoside option. I.e, the right page / uneven number
% 	will be indented by a small margin. This isn't done from the start
% 	when using the option which is unclear why. Changing it poses no
%	danger to the rest of the document.
%
%
% - Tested and working on following environments: Linux (debian) running texlive, 
%	Windows 10 running MikTeX, Overleaf.
%
%
% - Should the frontpage by any means overflow into the second page or
% 	similar. There exists a \*vspace that can be decreased to tighten the
%	space between upper and lower part of the frontpage. It is by default 
%	6cm and is located directly after \end{sloppypar} below.

\newfontfamily{\Georgiafont}{Georgia}
\newfontfamily{\ArialFont}{Arial}

\newcommand{\frontpage}[7]{
	\def\thesistitle			{#1}
	\def\thesissubtitle			{#2}
	\def\thesisswedishtitle		{#3}
	\def\thesisswedishsubtitle	{#4}
	\def\firstauthor			{#5}
	\def\secondauthor			{#6}
	
	\pagetitlecontinued#7
}

\newcommand{\pagetitlecontinued}[7]{
	\newgeometry{left=0.723in, right=0.723in, top=0.59in, bottom=0.5in}
	\thispagestyle{empty}
	\begin{spacing}{1}
		%logo
		{
			\hspace{-0.95cm}
			\includegraphics[width=0.208\textwidth]{kaulogo}
			\vspace{0.7cm}
		}
		\\
		\begin{sloppypar}
			% Title
			\ifthenelse{\equal{\thesistitle}{}}
			{
				\ifthenelse{\equal{#7}{English}}{{\fontsize{32pt}{32pt}\selectfont\Georgiafont\noindent Thesis title, set to the left and in Berthold Garamond or Georgia}}{{\fontsize{32pt}{32pt}\selectfont\Georgiafont\noindent Uppsatsens titel, vänsterställd och satt i Berthold Garamond alternativt Georgia}}
			}% If thesistitle is not empty, display parameter value instead.
			{{\fontsize{32pt}{32pt}\selectfont\Georgiafont\noindent \thesistitle}}
			\vspace*{0.9cm}\\
			{
				\fontsize{16pt}{16pt}\selectfont
				% Subtitle
				\ifthenelse{\equal{\thesissubtitle}{}}
				{
					\ifthenelse{\equal{#7}{English}}{{\selectfont\Georgiafont\noindent Thesis subtitle, set in Berthold Garamond or Georgia}}{{\selectfont\Georgiafont\noindent Uppsatsens undertitel på en rad i Berthold Garamond alternativ Georgia}}
				}% If thesissubtitle is not empty, display parameter value instead.
				{{\selectfont\Georgiafont\noindent \thesissubtitle}}
				\vspace*{0.05cm}\\
				\hspace*{-0.3cm}
				\rule{17.8cm}{1pt}
				\vspace*{-0.15cm}\\
				% Swe/Eng title
				\ifthenelse{\equal{\thesisswedishtitle}{}}
				{
					\ifthenelse{\equal{#7}{English}}{{\selectfont\Georgiafont Thesis Swedish title, set in Berthold Garamond or Georgia}}{{\selectfont\Georgiafont Uppsatsens engelska titel, satt i Berthold Garamond alternativt Georgia}}
				}% If thesisswedishtitle is not empty, display parameter value instead.
				{{\selectfont\Georgiafont \thesisswedishtitle}}
				\vspace*{0.1cm}\\
				% Swe/Eng subtitle
				\ifthenelse{\equal{\thesisswedishsubtitle}{}}
				{
					\ifthenelse{\equal{#7}{English}}{{\selectfont\Georgiafont Thesis subtitle in Swedish}}{{\selectfont\Georgiafont Uppsatsens undertitel på engelska}}
				}% If thesisswedishtitle is not empty, display parameter value instead.
				{{\selectfont\Georgiafont \thesisswedishsubtitle}}
				%				{\selectfont\Georgiafont \thesisswedishsubtitle}
				\vspace*{0.0cm}\\
				\hspace*{-0.3cm}
				\rule{17.8cm}{1pt}
				\vspace*{0.15cm}\\
				% First author
				\ifthenelse{\equal{\firstauthor}{}}
				{
					\ifthenelse{\equal{#7}{English}}{{\selectfont\Georgiafont First author}}{{\selectfont\Georgiafont Första författaren}}
				}
				{{\selectfont\Georgiafont \firstauthor}}
				\vspace*{0.1cm}\\
				% Second author
				\ifthenelse{\equal{\secondauthor}{}}
				{
					\ifthenelse{\equal{#7}{English}}{{\selectfont\Georgiafont Second author}}{{\selectfont\Georgiafont Andra författaren, lämna blank om ingen.} \vspace*{-1cm}} % the vspace is a fix for the Swedish being too big and growing into a second page.
				}
				{{\selectfont\Georgiafont \secondauthor}}
				
				\hspace*{-0.0cm}
			}
		\end{sloppypar}
	
%		If the upper part of the fronntpage get's too large, decrease this vspace to keep it from going over to a new page.
		\vspace*{6cm}
		{
			\fontsize{10pt}{\baselineskip}\selectfont
			\hspace*{-0.75cm}
			{\ArialFont\noindent #1}
			\vspace*{-0.3cm}\\
			\hspace*{-0.3cm}
			\rule{17.8cm}{1pt}
			\vspace*{-0.55cm}\\
			{\ArialFont\noindent #2}
			\vspace*{-0.3cm}\\
			\hspace*{-0.3cm}
			\rule{17.8cm}{1pt}
			\vspace*{-0.55cm}\\
			{\ArialFont\noindent #3}
			\vspace*{-0.3cm}\\
			\hspace*{-0.3cm}
			\rule{17.8cm}{1pt}
			\vspace*{-0.55cm}\\
			\ifthenelse{\equal{#7}{English}}{{\ArialFont\noindent Supervisor: #4}}{{\ArialFont\noindent Handledare: #4}}
			\vspace*{-0.3cm}\\
			\hspace*{-0.3cm}
			\rule{17.8cm}{1pt}
			\vspace*{-0.55cm}\\
			\ifthenelse{\equal{#7}{English}}{{\ArialFont\noindent Examiner: #5}}{{\ArialFont\noindent Examinator: #5}}
			\vspace*{-0.3cm}\\
			\hspace*{-0.3cm}
			\rule{17.8cm}{1pt}
			\vspace*{-0.55cm}\\
			\ifthenelse{\equal{#7}{English}}{{\ArialFont\noindent Date: #6}}{{\ArialFont\noindent Datum: #6}}
			\vspace*{-0.3cm}\\
			\hspace*{-0.3cm}
			\rule{17.8cm}{1pt}
			\vspace*{-0.55cm}\\
		}
	\end{spacing}
	\clearpage
	\restoregeometry
	
	% Empty page with no numbering after frontpage.
	\thispagestyle{empty}
	\mbox{}
	\newpage
}
